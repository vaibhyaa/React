# ✅ 2) `useBookings.js` → Custom Hook

```js id="bo4zax"
import { useQuery } from "@tanstack/react-query";
import { useSearchParams } from "react-router-dom";
import { getBookings } from "../../services/apiBookings";

export function useBookings() {
  // 1️⃣ Read URL query params
  // Example URL:
  // /bookings?status=checked-in&sortBy=startDate-desc
  const [searchParams] = useSearchParams();

  // -----------------------------
  // FILTER LOGIC
  // -----------------------------

  // 2️⃣ Get 'status' from URL
  // If no status is found, default = "all"
  const filterValue = searchParams.get("status") || "all";

  // 3️⃣ If status = "all", don't apply filter
  // Otherwise create filter object
  const filter =
    !filterValue || filterValue === "all"
      ? null
      : { field: "status", value: filterValue };

  // Example:
  // filter = { field: "status", value: "checked-in" }

  // -----------------------------
  // SORT LOGIC
  // -----------------------------

  // 4️⃣ Get sortBy from URL
  // If no sortBy exists, default = "startDate-desc"
  const sortByRaw = searchParams.get("sortBy") || "startDate-desc";

  // 5️⃣ Split string into field + direction
  const [field, direction] = sortByRaw.split("-");

  // 6️⃣ Create sort object
  const sortBy = { field, direction };

  // Example:
  // sortBy = { field: "startDate", direction: "desc" }

  // -----------------------------
  // REACT QUERY
  // -----------------------------

  // 7️⃣ useQuery runs immediately when hook runs
  // queryKey helps React Query cache data
  // If filter or sortBy changes, queryKey changes -> data refetches
  const { isLoading, data, error } = useQuery({
    queryKey: ["bookings", filter, sortBy],

    // 8️⃣ queryFn is the function React Query will call
    // This is where getBookings() is triggered
    queryFn: () => getBookings({ filter, sortBy }),
  });

  // -----------------------------
  // SAFE DATA EXTRACTION
  // -----------------------------

  // 9️⃣ Initially data is undefined while loading
  // So use optional chaining to avoid crash
  const bookings = data?.data || [];
  const count = data?.count || 0;

  //  🔟 Return values to component
  return { isLoading, bookings, error, count };
}
```

# ✅ 3) `apiBookings.js` → API / Supabase Service Function

```js id="d9plfc"
import supabase from "./supabase";

export async function getBookings({ filter, sortBy }) {
  // 1️⃣ Start building Supabase query
  // Select all columns from bookings table
  // Also fetch related cabins and guests data
  // count: "exact" gives total number of matching rows
  let query = supabase
    .from("bookings")
    .select("*, cabins(*), guests(*)", { count: "exact" });

  // -----------------------------
  // APPLY FILTER
  // -----------------------------

  // 2️⃣ If filter exists, apply it
  // Example:
  // filter = { field: "status", value: "checked-in" }
  // This becomes:
  // .eq("status", "checked-in")
  if (filter) {
    query = query[filter.method || "eq"](filter.field, filter.value);
  }

  // -----------------------------
  // APPLY SORTING
  // -----------------------------

  // 3️⃣ If sortBy exists, apply sorting
  // Example:
  // sortBy = { field: "startDate", direction: "desc" }
  if (sortBy) {
    query = query.order(sortBy.field, {
      ascending: sortBy.direction === "asc",
    });
  }

  // -----------------------------
  // EXECUTE QUERY
  // -----------------------------

  // 4️⃣ await query actually sends request to Supabase database
  const { data, error, count } = await query;

  // 5️⃣ If DB returns error, throw error
  if (error) {
    console.error(error);
    throw new Error("Bookings not found");
  }

  // 6️⃣ Return result back to useQuery()
  return { data, count };
}
```

# ✅ Small Important Fix for Your `Table.Body`

Your `BookingTable` uses:

```jsx id="2hlfzd"
<Table.Body data={bookings} />
```

So your reusable `Table.Body` should be:

```js id="t6s0j6"
function Body({ data, render }) {
  if (data.length === 0) {
    return <Empty>No data to show at the Moment</Empty>;
  }

  return <StyledBody>{data.map(render)}</StyledBody>;
}
```

❌ Not:

```js id="vw4tze"
function Body({ cabinsAllData, render })
```

Because now it should work for **cabins, bookings, users, etc.**
